
//Lambda Function

console.log('Loading contact form processing function.');
// Load the SES API.
var AWS = require('aws-sdk');
var ses = new AWS.SES();

exports.handler = function (event, context) {
    console.log('Received event:', JSON.stringify(event, null, 2));

    // Check that we received all the required data.
    if (!event.name || event.name.trim() === '') {
        context.fail('Please complete the name field.');
        return;
    }
    
    if (!event.email) {
        context.fail('Please add your email address.');
        return;
    }
    
    if (!event.subject || event.subject.trim() === '') {
        context.fail('Please add a subject.');
        return;
    }
    
    if (!event.message || event.message.trim() === '') {
        context.fail('Did you forget a message?');
        return;
    }

    // Put together all info needed to send the email.
    var name = event.name.trim();
    var email = unescape(event.email.trim());
    var replyTo = event.name + " <" + email + ">";
    var subject = event.subject.trim();
    var message = "Incoming message from " + name + " <" + email + ">\n\n" + event.message.trim();

    // Send the email via SES.
    var params = {
        Destination: { ToAddresses: [ 'Your Target Email Address' ] },
        Message: {
            Body: { Text: { Data: message, Charset: 'UTF-8' } },
            Subject: { Data: subject, Charset: 'UTF-8' }
        },
        Source: "Your Source Email Address",
        ReplyToAddresses: [ replyTo ]
    };
    ses.sendEmail(params, function (err, data) {
        if (err) {
            console.log(err, err.stack);
            context.fail("Hmm something didn't work. Please try again. Thanks!");
            return;
        } else {
            console.log(data);
            context.succeed("Your message has been sent - I'll get back to you soon. Thanks!");
            return;
        }
    });
};
